### Create a new user
curl http://localhost:3000/users/create --request POST -H "content-type: application/json" --data '{"username": "newUser2", "email": "newUser2@gmail.com", "password": "secure123441", "repeatPassword": "secure123441"}'

### Check if user has 2FA enabled
curl --request POST http://localhost:3000/users/preLogin -H "content-type: application/json" --data '{"email": "newUser2@gmail.com", "password": "secure123441"}'

### Login user
curl --request POST http://localhost:3000/users/login -H "content-type: application/json" --data '{"email": "newUser2@gmail.com", "password": "secure123441"}'

### Login user with 2FA
curl --request POST http://localhost:3000/users/login2FA -H "content-type: application/json" --data '{"email": "newUser2@gmail.com", "password": "secure123441", "code": "152436"}'
